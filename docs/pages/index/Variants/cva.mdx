```tsx title="StyledAlert.tsx"
import { type ComponentPropsWithoutRef } from "react"
import { cva, type VariantProps } from "class-variance-authority"

const alertVariants = cva("p-4 rounded-md", {
  variants: {
    severity: {
      warning: "bg-warning/20 text-warning",
      info: "bg-info/20 text-info",
      error: "bg-error/20 text-error",
    },
    isActive: {
      true: "custom-class-active border border-current/50 shadow-xl shadow-current/20",
      false: "custom-class-inactive border border-current/20 border-dashed",
    },
  },
  defaultVariants: {
    severity: "info",
    isActive: false,
  },
})

type AlertProps = ComponentPropsWithoutRef<"div"> &
  VariantProps<typeof alertVariants>

const Alert = ({
  severity,
  isActive,
  className,
  ...props
}: AlertProps) => {
  return (
    <div
      {...props}
      className={alertVariants({ severity, isActive, className })}
    />
  )
}
```
