```ts title="StyledAlert.ts"
import ma from "@marmo/react"

interface AlertExtraProps {
  $isActive?: boolean
  $ringColor?: string
}

interface AlertVariantProps {
  $severity?: "info" | "warning" | "error"
  $outlined?: boolean
}

const Alert = ma.div.variants<AlertExtraProps, AlertVariantProps>({
  base: ({ $isActive }) =>
    `p-4 rounded-md ${$isActive ? "shadow-lg" : ""}`,
  variants: {
    $severity: {
      info: "bg-info text-info-content",
      warning: "bg-warning text-warning-content",
      error: ({ style, $ringColor }) =>
        `bg-error text-error-content ${style({ outlineColor: $ringColor })}`,
    },
    $outlined: {
      true: "outline outline-2",
      false: "outline-none",
    },
  },
  defaultVariants: {
    $severity: "info",
    $outlined: false,
  },
})
```
