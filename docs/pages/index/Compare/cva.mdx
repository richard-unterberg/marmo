```tsx title="MyAlert.tsx"
import { type ComponentPropsWithoutRef, type CSSProperties, forwardRef } from "react"
import { cva, type VariantProps } from "class-variance-authority"

const alertVariants = cva("p-4 rounded-md", {
  variants: {
    $severity: {
      warning: "bg-yellow-100 text-yellow-800 dark:bg-yellow-600/50 dark:text-yellow-200",
      info: "bg-blue-100 text-blue-800 dark:bg-blue-600/50 dark:text-blue-200",
      error: "bg-red-100 text-red-800 dark:bg-red-600/50 dark:text-red-200",
    },
    $isActive: {
      true: "custom-class-active",
      false: "custom-class-inactive border-gray-300 shadow-sm",
    },
  },
  defaultVariants: {
    $severity: "info",
    $isActive: false,
  },
})

interface AlertProps
  extends ComponentPropsWithoutRef<"div">,
    VariantProps<typeof alertVariants> {}

const activeStyle = {
  border: "1px solid color-mix(in oklab, currentColor 18%, transparent)",
  boxShadow: "0 14px 40px color-mix(in oklab, currentColor 18%, transparent)",
} satisfies CSSProperties

const Alert = forwardRef<HTMLDivElement, AlertProps>(
  (
    {
      $severity,
      $isActive,
      className,
      style,
      ...props
    },
    ref,
  ) => {
    return (
      <div
        {...props}
        ref={ref}
        className={alertVariants({ $severity, $isActive, className })}
        style={{
          ...($isActive ? activeStyle : undefined),
          ...style,
        }}
      />
    )
  },
)

```
